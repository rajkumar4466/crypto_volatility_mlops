**`src/ingestion/coingecko.py`:**
```python
import requests
import boto3
import json
import io
import pandas as pd
from datetime import datetime, timezone

COINGECKO_OHLCV_URL = "https://api.coingecko.com/api/v3/coins/bitcoin/ohlcv"
HEADERS = {"User-Agent": "crypto-volatility-mlops/1.0"}

def fetch_ohlcv(days: float = 0.1) -> list[list]:
    """Fetch BTC 1-min OHLCV candles from CoinGecko (no API key required).
    days=0.1 ≈ 144 minutes of 1-min candles on the free tier.
    Raises ValueError on any null field — do not propagate NaN into features.
    Rate limit: 30 req/min. One call per DAG run is safe.
    """
    resp = requests.get(
        COINGECKO_OHLCV_URL,
        params={"vs_currency": "usd", "days": days},
        headers=HEADERS,
        timeout=30,
    )
    resp.raise_for_status()
    candles = resp.json()  # [[ts_ms, open, high, low, close, volume], ...]
    # Guard against null fields — raise rather than propagate NaN
    for i, row in enumerate(candles):
        if any(v is None for v in row):
            raise ValueError(f"Null value in CoinGecko candle row {i}: {row}")
    return candles

def candles_to_dataframe(candles: list[list]) -> pd.DataFrame:
    """Convert raw CoinGecko candle list to DataFrame."""
    df = pd.DataFrame(candles, columns=["timestamp", "open", "high", "low", "close", "volume"])
    df = df.sort_values("timestamp").reset_index(drop=True)
    return df

def write_raw_to_s3(df: pd.DataFrame, bucket: str, key: str) -> None:
    """Write raw OHLCV DataFrame to S3 as Parquet."""
    buf = io.BytesIO()
    df.to_parquet(buf, index=False)
    buf.seek(0)
    s3 = boto3.client("s3")
    s3.upload_fileobj(buf, bucket, key)
```

```python
import pandas as pd
import numpy as np

FEATURE_COLS = [
    "volatility_10m", "volatility_30m", "volatility_ratio",
    "rsi_14",
    "volume_spike", "volume_trend",
    "price_range_30m", "sma_10_vs_sma_30", "max_drawdown_30m",
    "candle_body_avg",
    "hour_of_day", "day_of_week",
]

def compute_features(df: pd.DataFrame) -> pd.DataFrame:
    """Compute all 12 features. min_periods enforced on every rolling call.
    Rows where any feature is NaN (warm-up period) must be dropped by the caller
    before writing to Feast. This function does NOT drop them — callers decide
    whether to drop or retain for diagnostics.

    WARNING: Do NOT call this on test data that includes future rows.
    Fit scalers on train split ONLY.
    """
    df = df.copy().sort_values("timestamp").reset_index(drop=True)

    ret = df["close"].pct_change()

    # Volatility (std of returns over window)
    df["volatility_10m"] = ret.rolling(10, min_periods=10).std()
    df["volatility_30m"] = ret.rolling(30, min_periods=30).std()
    # Ratio: avoid division by zero with replace
    vol30_safe = df["volatility_30m"].replace(0, np.nan)
    df["volatility_ratio"] = df["volatility_10m"] / vol30_safe

    # RSI-14 via EWM (standard Wilder smoothing approximation)
    delta = df["close"].diff()
    gain = delta.clip(lower=0).ewm(span=14, min_periods=14, adjust=False).mean()
    loss = (-delta.clip(upper=0)).ewm(span=14, min_periods=14, adjust=False).mean()
    loss_safe = loss.replace(0, np.nan)
    df["rsi_14"] = 100 - (100 / (1 + gain / loss_safe))

    # Volume features
    vol_mean_30 = df["volume"].rolling(30, min_periods=30).mean()
    vol_mean_30_safe = vol_mean_30.replace(0, np.nan)
    df["volume_spike"] = df["volume"] / vol_mean_30_safe
    vol_mean_10 = df["volume"].rolling(10, min_periods=10).mean()
    df["volume_trend"] = vol_mean_10 / vol_mean_30_safe

    # Price features
    df["price_range_30m"] = (
        df["high"].rolling(30, min_periods=30).max()
        - df["low"].rolling(30, min_periods=30).min()
    )
    sma10 = df["close"].rolling(10, min_periods=10).mean()
    sma30 = df["close"].rolling(30, min_periods=30).mean()
    sma30_safe = sma30.replace(0, np.nan)
    df["sma_10_vs_sma_30"] = sma10 / sma30_safe
    rolling_max = df["close"].rolling(30, min_periods=30).max()
    rolling_max_safe = rolling_max.replace(0, np.nan)
    df["max_drawdown_30m"] = (df["close"] - rolling_max_safe) / rolling_max_safe
    df["candle_body_avg"] = (df["close"] - df["open"]).abs().rolling(10, min_periods=10).mean()

    # Temporal features (no leakage risk — derived from current candle timestamp)
    dt = pd.to_datetime(df["timestamp"], unit="ms")
    df["hour_of_day"] = dt.dt.hour.astype(int)
    df["day_of_week"] = dt.dt.dayofweek.astype(int)

    return df
```

**`src/features/labels.py`:**
```python
import pandas as pd

SWING_THRESHOLD = 0.02  # >2% price swing in next 30 minutes = VOLATILE

def label_volatility(df: pd.DataFrame) -> pd.DataFrame:
    """Label each row as VOLATILE or CALM based on future 30-minute price swing.

    Label at row T = max |close[T+1..T+30] - close[T]| / close[T] > 2%

    CRITICAL: Label uses ONLY rows T+1 through T+30 (forward window).
    Row T is NOT included in the future window.
    Last 30 rows are dropped — they have no complete forward window.

    This function is the look-ahead bias gate. Do NOT modify the slice to
    include row T itself (i:i+30 instead of i+1:i+31 would be a bug).
    """
    df = df.copy().sort_values("timestamp").reset_index(drop=True)
    n = len(df)
    labels = []

    for i in range(n):
        # Forward window: T+1 to T+30 (exclusive of T itself)
        end = min(i + 31, n)  # i+31 so slice [i+1:i+31] covers T+1..T+30
        if end - (i + 1) < 30:
            labels.append(float("nan"))  # incomplete future window
            continue
        future_prices = df["close"].iloc[i + 1 : i + 31]
        current_price = df["close"].iloc[i]
        if current_price == 0:
            labels.append(float("nan"))
            continue
        swing = (future_prices.max() - future_prices.min()) / current_price
        labels.append("VOLATILE" if swing > SWING_THRESHOLD else "CALM")

    df["label"] = labels
    # Drop rows with incomplete forward window (last 30 rows)
    df = df.dropna(subset=["label"]).reset_index(drop=True)
    return df
```

```python
"""
Standalone ingest script. Run directly or call from Airflow PythonOperator.
Usage: python -m src.ingestion.ingest --bucket my-bucket --prefix raw/btc_ohlcv/
"""
import argparse
import os
from datetime import datetime, timezone
from src.ingestion.coingecko import fetch_ohlcv, candles_to_dataframe, write_raw_to_s3

def main(bucket: str, prefix: str, days: float = 0.1):
    print(f"Fetching BTC OHLCV from CoinGecko (days={days})...")
    candles = fetch_ohlcv(days=days)
    df = candles_to_dataframe(candles)
    print(f"Fetched {len(df)} candles. Range: {df['timestamp'].min()} to {df['timestamp'].max()}")
    ts = datetime.now(timezone.utc).strftime("%Y%m%dT%H%M%SZ")
    key = f"{prefix}{ts}.parquet"
    write_raw_to_s3(df, bucket, key)
    print(f"Written to s3://{bucket}/{key}")
    return df

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--bucket", required=True)
    parser.add_argument("--prefix", default="raw/btc_ohlcv/")
    parser.add_argument("--days", type=float, default=0.1)
    args = parser.parse_args()
    main(args.bucket, args.prefix, args.days)
```

```python
"""
Feature engineering pipeline. Called by Feast store.py or Airflow task.
Produces labeled, feature-complete DataFrame ready for Feast offline write.
"""
import pandas as pd
import numpy as np
from src.features.compute import compute_features, FEATURE_COLS
from src.features.labels import label_volatility

def run_feature_pipeline(raw_df: pd.DataFrame) -> dict:
    """
    Input: raw OHLCV DataFrame from CoinGecko ingest.
    Output: dict with:
      - "full": complete labeled+featured df (used for Feast offline write)
      - "train": time-ordered training split (first 80%)
      - "test": time-ordered test split (last 20%)

    NOTE on look-ahead bias:
    - min_periods enforced on all rolling windows (see compute.py)
    - labels use T+1..T+30 forward window only (see labels.py)
    - split is by index cutoff — NEVER shuffle time series data
    - training accuracy > 80% on 288 samples likely indicates leakage — investigate
    """
    # Step 1: compute features
    featured_df = compute_features(raw_df)

    # Step 2: drop NaN rows from rolling warm-up (first ~30 rows)
    featured_df = featured_df.dropna(subset=FEATURE_COLS).reset_index(drop=True)

    # Step 3: apply VOLATILE/CALM labels (drops last 30 rows)
    labeled_df = label_volatility(featured_df)

    # Step 4: time-ordered split — NEVER shuffle
    cutoff = int(len(labeled_df) * 0.8)
    train_df = labeled_df.iloc[:cutoff].copy()
    test_df  = labeled_df.iloc[cutoff:].copy()

    assert len(train_df) > 0 and len(test_df) > 0, "Insufficient data for train/test split"
    if len(train_df) > 0 and len(test_df) > 0:
        assert train_df["timestamp"].max() < test_df["timestamp"].min(), \
            "Train/test split is NOT time-ordered — data contamination detected"

    return {
        "full": labeled_df,
        "train": train_df,
        "test": test_df,
    }
```
